start hi_there;

func hi_there() -> int {
    char c = 'a';
    println("c = ", 'a');

    int x = 7 + 6 * (12 / 3! - 10) + 42 + 1;
    bismach();
    println();
    test();

    return 0;
}

func bismach() {
    int x = 1;
    int y = 2;
    println("x = ", x, '\t', "y = ", y);
    swap x, y;
    println("x = ", x, '\t', "y = ", y);
}

func test() -> void {
    loop i: 1 to 6 times {
        loop i times {
            print("*");
        }
        println();
    }
}
